d3 real-world implementation

  1. load data
  2. make a selection
  3. apply data
  4. style with data
  5. add actions

1. load data


d3.text("/static/hedonometer/data/labMT/labMTscores-"+lang+".csv", function (text) {
    var tmp = text.split("\n");
    hedotools.lens = tmp.map(parseFloat);
    var len = lens.length - 1;
    while (!lens[len]) {
        lens = lens.slice(0, len);
        len--;
    }
    if (!--csvLoadsRemaining) hedotools.initializePlot();
});

2. make a selection


axes.selectAll("rect.shiftrect");

3. apply data


axes.selectAll("rect.shiftrect")
    .data(sortedMag);

4. style with data


axes.selectAll("rect.shiftrect")
    .data(sortedMag)
    .enter()
    .append("rect")
    .attr({ 
	"class": function(d,i) { return "shiftrect "+intStr[sortedType[i]]+" "+typeClass[sortedType[i]]; },
	"x": function(d,i) { 
	    if (d>0) { return figcenter; } 
	    else { return bigshiftx(d)}
	},
	"y": function(d,i) { return bigshifty(i+1); },
	"height": function(d,i) { return iBarH; },
	"width": function(d,i) { 
	    if ((d)>0) { return bigshiftx(d)-bigshiftx(0); }
	    else { return bigshiftx(0)-bigshiftx(d); } 
	},
	"opacity": "0.7",
	"stroke-width": "1",
	"stroke": "rgb(0,0,0)"
    });

5. add actions


axes.selectAll("rect.shiftrect")
    .data(sortedMag)
    .enter()
    .append("rect")
    .attr({ 
	"class": function(d,i) { return "shiftrect "+intStr[sortedType[i]]+" "+typeClass[sortedType[i]]; },
	"x": function(d,i) { 
	    if (d>0) { return figcenter; } 
	    else { return bigshiftx(d)}
	},
	"y": function(d,i) { return bigshifty(i+1); },
	"height": function(d,i) { return iBarH; },
	"width": function(d,i) { 
	    if ((d)>0) { return bigshiftx(d)-bigshiftx(0); }
	    else { return bigshiftx(0)-bigshiftx(d); } 
	},
	"opacity": "0.7",
	"stroke-width": "1",
	"stroke": "rgb(0,0,0)"
    })
    .on("mouseover", function(d){
        var rectSelection = d3.select(this).style({opacity:"1.0"});
    })
    .on("mouseout", function(d){
         var rectSelection = d3.select(this).style({opacity:"0.7"});
    });

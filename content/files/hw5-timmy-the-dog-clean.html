function plantTrees(dim::Int,p::Float64)
    f = zeros(Bool,dim,dim)
    for i in 1:length(f)
        if rand() < p
            f[i] = true
        end
    end
    f
end

function scanline(labeled,f,i,j,k,l,c)
    # in place modify labeled
    # with current label k
    labeled[i,j] = k
    c[1]+=1
    
    # scan up, down, then pass to left right if it can

    # scan up
    u = copy(i)-1
    while u >= 1
        if f[u,j] && labeled[u,j] == 0
            labeled[u,j] = k
            c[1]+=1
        else
            break
        end
        u-=1
    end
    # scan down
    d = copy(i)+1
    while d <= l
        if f[d,j] && labeled[d,j] == 0
            labeled[d,j] = k
            c[1]+=1
        else
            break
        end
        d+=1
    end
    
    # check on the right
    if j < l
        for r=flipud((u+1):i)
            if f[r,j+1] && (labeled[r,j+1] == 0)
                scanline(labeled,f,r,j+1,k,l,c)
            end
        end
        for r=(i+1):(d-1)
            if f[r,j+1] && (labeled[r,j+1] == 0)
                scanline(labeled,f,r,j+1,k,l,c)
            end
        end
    end
    if j > 1
        for r=flipud(min(i,u+1):i)
            if f[r,j-1] && (labeled[r,j-1] == 0)
                scanline(labeled,f,r,j-1,k,l,c)
            end
        end
        for r=i+1:min(d-1,l)
            if f[r,j-1] && (labeled[r,j-1] == 0)
                scanline(labeled,f,r,j-1,k,l,c)
            end
        end
    end
    return d+1,j
end

function conncomp2(f::Array{Bool,2})
    # implement a flood fill with scanline, for speed
    # this is what matlab's bwconncomp does
    l = size(f)[1]
    labeled = zeros(Int64,l,l)
    k=0
    j=0
    i=0
    compsize = Int64[]
    while j<l
        j+=1
        i=0
        while i<l
            i+=1
            if labeled[i,j] == 0 && f[i,j]
                k+=1
                c=[0]
                i,j = scanline(labeled,f,i,j,k,l,c)
                push!(compsize,c[1])
            end
        end
    end
    labeled,compsize
end
